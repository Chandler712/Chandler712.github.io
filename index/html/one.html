    # canvas练习

    <!DOCTYPE html>
<head>
    <meta charset="utf-8">
    <title>clock</title>
    <style>
    .clocks{
        height:500px;
        margin: 25px auto;
        position: relative;
        width:500px;
        }
    /*加个背景*/
    #canvas{
        background-color: rgb(214, 212, 198);
    }
</style>
</head>
<body>
<div class="clocks">
    <canvas id="canvas" width="500" height="500"></canvas>
</div>
</body>
</html>
<script>
//声明变量
var canvas,ctx;
var clockRadius=250;
var clockImage;
//画图函数
function clear(){//清除指定画布
    ctx.clearRect(0,0,ctx.canvas.width,ctx.canvas.height);
}
function drawScene(){
    clear();
    //获取当前时间
    var date=new Date();
    var hours=date.getHours();
    var minutes=date.getMinutes();
    var seconds=date.getSeconds();
    //hours=hours>12?hours-12:hours;
    if(hours>12){ //Date()是24小时制，表盘12小时制
        hours=hours-12;
    }else{
        hours=hours;
    };
    var hour=hours+minutes/60;
    var minute=minutes+seconds/60;
    //保存当前内容
    ctx.save();
    //画钟盘背景图，点线圆圈图
    ctx.drawImage(clockImage,0,0,500,500);
    ctx.translate(canvas.width/2,canvas.height/2);//画布轴心坐标变成（250,250）
    ctx.beginPath();

    //绘制时钟数字
    ctx.font="40px Arial";//设置显示刻度的数字 1,2,3.. 的字体及字号 
    ctx.fillStyle="#000";
    ctx.textAlign="center";//绘制的刻度在整个画布左右居中 
    ctx.textBaseline="middle";//绘制的刻度在整个画布上下居中 
    for(var n=1;n<=12;n++){
        var theta=(n-3)*(Math.PI*2)/12; //绘制1-12数字的位置
        var x=clockRadius*0.7*Math.cos(theta);//确定各刻度点 X、Y坐标 
        var y=clockRadius*0.7*Math.sin(theta);
        ctx.fillText(n,x,y); //绘制文字 当n=1时，theta=-PI/3 绘制的刻度在整个画布左右居中 

    };
    //画时针
    ctx.save();//存储画布状态
    var theta=(hour-3)*2*Math.PI/12; //旋转间隔是30度
    ctx.rotate(theta);
    ctx.beginPath();//开始一条路径
    ctx.moveTo(-15,-5);
    ctx.lineTo(-15,5);
    ctx.lineTo(clockRadius*0.5,1);
    ctx.lineTo(clockRadius*0.5,-1);
    ctx.fillStyle="#030303";
    ctx.fill();
    ctx.restore();
    //画分针
    ctx.save();
    var theta=(minute-15)*2*Math.PI/60;
    ctx.rotate(theta);
    ctx.beginPath();
    ctx.moveTo(-15,-4);
    ctx.lineTo(-15,4);
    ctx.lineTo(clockRadius*0.8,1);
    ctx.lineTo(clockRadius*0.8,-1);
    ctx.fillStyle="#A0522D";
    ctx.fill();
    ctx.restore();
    //画秒针
    ctx.save();
    var theta=(seconds-15)*2*Math.PI/60;
    ctx.rotate(theta);
    ctx.beginPath();
    ctx.moveTo(-15,-3);
    ctx.lineTo(-15,3);
    ctx.lineTo(clockRadius*0.9,1);
    ctx.lineTo(clockRadius*0.9,-1);
    ctx.fillStyle="#F3A829";
    ctx.fill();
    ctx.restore();
    ctx.restore();
}
//初始化
setInterval(function(){
    canvas=document.getElementById("canvas");
    ctx=canvas.getContext("2d");
    
    clockImage=new Image();
    clockImage.src="https://static.runoob.com/images/mix/125855_nnla_89964.png";//图片
    setInterval(drawScene,1000);
})
</script>
